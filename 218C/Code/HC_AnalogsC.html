#include "HC_Analogs.h"

// Calibrations

const static int MicThresh  = 80; //0-1023

const static int AccelLevelPos = 533;

const static int AccelDeadBand = 5;

const static float AccelGain = 2.0;

const static int PotNeutralPos = 907;

const static int PotDeadBand = 25;

const static float PotGain = -1.5;

const static int MinMotorDuty = 40; //in percent, so 40%

 

unsigned char Motor0=0;

unsigned char Motor1=0;

unsigned int myPot = 0;

unsigned int accelSignal=0;

int M0Duty = 0;

int M1Duty = 0;

 

/*----------- Public Prototypes ----------*/

void init_HC_Analogs(void);

void MotorCmds(void);

unsigned char checkMic(void);

unsigned char Get_Motor0(void);

unsigned char Get_Motor1(void);

 

 

/*------------ Code --------------*/

void init_HC_Analogs(void){

  (void)ADS12_Init("AAAAAAAA");

 

}

 

unsigned char checkMic(void){

  unsigned int micSignal;

  micSignal = ADS12_ReadADPin(2);

 

  if (micSignal > MicThresh){

    return 0xf1; //blow airhorn

  }

  else{

    return 0xf0; //don't blow airhorn

  }

 

}

 

 

void MotorCmds (void){

  static char PotInDeadZone = 1;

  static char AccelInDeadZone = 1;

 

  int offset = 0;

  int diff = 0;

  int myInputCmd = 0;

  int rawM0 = 0;

  int rawM1 = 0;

 

  unsigned int Dir0=0;

  unsigned int Dir1=0;

  unsigned int PWM0=0;

  unsigned int PWM1=0;

 

  // read inputs

  myPot = ADS12_ReadADPin(0);

  //(void)printf("pot: %i\n\r",(unsigned int)myPot);

  accelSignal = ADS12_ReadADPin(1);

 

  // calc diff from accelerometer

  if (accelSignal > AccelLevelPos) {   //turn left

    myInputCmd = (accelSignal - AccelLevelPos);

    if (myInputCmd > AccelDeadBand) diff = (int)(-1*AccelGain*myInputCmd);

    else offset = 0;

  }

  else { //turn right

    myInputCmd = (AccelLevelPos - accelSignal);

    if (myInputCmd > AccelDeadBand) diff = (int)(AccelGain*myInputCmd);

    else offset = 0;

  }

 

  // calc offset from pot

  if (myPot > PotNeutralPos) {   //forward

    myInputCmd = (myPot - PotNeutralPos);

    if (myInputCmd > PotDeadBand) offset = (int)(PotGain*myInputCmd);

    else offset = 0;

  }

  else { //backward

    myInputCmd = (PotNeutralPos - myPot);

    if (myInputCmd > PotDeadBand) offset = (int)(-1*PotGain*myInputCmd);

    else offset = 0;

  }

 

  // combine into raw duty cycle commands

  rawM0 = offset + diff;

  rawM1 = offset - diff;

 

  //saturate to [-100, 100]

  if (rawM0 > 100) rawM0 = 100;

  if (rawM0 < -100) rawM0 = -100;

  if (rawM1 > 100) rawM1 = 100;

  if (rawM1 < -100) rawM1 = -100;

 

  //add deadband to require abs(duty cycle) > MinMotorDuty

  if ((rawM0 > 0) && (rawM0 < MinMotorDuty)) rawM0 = 0;

  if ((rawM0 < 0) && (rawM0 > -1*MinMotorDuty)) rawM0 = 0;

  if ((rawM1 > 0) && (rawM1 < MinMotorDuty)) rawM1 = 0;

  if ((rawM1 < 0) && (rawM1 > -1*MinMotorDuty)) rawM1 = 0;

 

  //save these to module variables

  M0Duty = rawM0;

  M1Duty = rawM1;

 

  //Motor0

  if (rawM0 > 0) Dir0 = 0;

  else {

    Dir0 = 1;

    rawM0 = -1*rawM0;    // convert to positive

  }

  PWM0 = (char)(((float)rawM0)/3.2);  //converts to 0-31 range

  Motor0 = PWM0 % 32;                 //confirm that PWM signal only takes up lowest 5 bits

  if (Dir0 == 1) Motor0 += 32;        //set dir bit

  //Motor0 += 0x00;                   //sets address

 

  //Motor1

  if (rawM1 > 0) Dir1 = 0;

  else {

    Dir1 = 1;

    rawM1 = -1*rawM1; // convert to positive

  }

  PWM1 = (char)(((float)rawM1)/3.2); //converts to 0-31 range

  Motor1 = PWM1 % 32;                //confirm that PWM signal only takes up lowest 5 bits

  if (Dir1 == 1) Motor1 += 32;       //set dir bit

  Motor1 += 0x40;                    //sets address

}

 

unsigned char Get_Motor0(void){

  return Motor0;

}

 

unsigned char Get_Motor1(void){

  return Motor1;

}

 

 

void HC_Anal_Debug(){

  int lastTime=0;

 

  init_HC_Analogs();

  Timer_Init();

 

  while(1){

 

    if ((Timer_GetTime() - lastTime) > 500){

      MotorCmds();

      (void)printf("Motor0Byte= %x\r\n" , Get_Motor0());

      (void)printf("Motor1Byte= %x\r\n" , Get_Motor1());

 

      (void)printf("Motor0Duty= %d\r\n", M0Duty);

      (void)printf("Motor1Duty= %d\r\n", M1Duty);

      (void)printf("myPot= %d  \r\n" , myPot);

      (void)printf("accelSignal= %d  \r\n\n\n" , accelSignal);

 

      lastTime = Timer_GetTime();

    }

  }

 

}
