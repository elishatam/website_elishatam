#include <stdio.h>

#include <hidef.h>      /* common defines and macros */

#include <mc9s12e128.h>     /* derivative information */

#include "LCD.h"

#include "ME218_E128.h"

#include "defs.h"

#include "XBee.h"

#include "timer.h"

#include "HC_Analogs.h"

 

#pragma LINK_INFO DERIVATIVE "SampleS12"

 

#define TRANSMIT_INTERVAL 200       //in ms

 

void Init(void) {

  //setup ports

  DDRT = 0xff;           //all port T pins to output

  DDRU = 0xff;           //all port U pins to output

  //AD PINS

}

 

void main(void) {

  unsigned int TXlasttime = 0;

  unsigned char TXstart = 1;

 

  Init();

  Timer_Init();             //1ms timer

  XBee_Init();

  LCD_Init();

  init_HC_Analogs();

  EnableInterrupts;

  //XBee_Debug();

  //HC_Anal_Debug();

  (void)printf("main\r\n");

  wait(2000);

  LCD_Wait_for_Team();

  for(;;) {

    //sample inputs at the TRANSMIT_INTERVAL RATE

    //if this is a new transmit time interval get the current time

    if(TXstart == 1) {

      TXlasttime = Timer_GetTime();

      TXstart = 0;

    } else {

      if((Timer_GetTime() - TXlasttime) > TRANSMIT_INTERVAL) {

        //get the bytes from the analog input controls

        //mimic for now

        MotorCmds();

        //send bytes to boat

        XBee_Transmit(Get_Motor0(),Get_Motor1(),checkMic());

        //restart time interval

        TXstart = 1;

      }

    }

    XBee_Receive();

  }

}
