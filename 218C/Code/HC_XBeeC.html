//Boat XBee = 0x208A

#define BOAT_ADDRESS_MSB 0x21

#define BOAT_ADDRESS_LSB 0x8A

#define XBEE_START_BYTE 0x7E

#define SEND_MESSAGE 0x01        //API identifyer for a sent message

#define PACKET_LENGTH 30         //predetermined based on protocal

#define DATA_BYTES 4             //predetermined based on protocal

static unsigned char receive_complete = 0;

static unsigned char transmit_packet[PACKET_LENGTH] = 0;

static unsigned char receive_packet[PACKET_LENGTH] = 0;

static unsigned char receive_buffer[PACKET_LENGTH] = 0;

static unsigned char transmit_index = 0;

static unsigned char receive_index = 0;

static unsigned char waterlogged = 0;

void XBee_Init(void) {

  (void)printf("sci init\r\n");

 

  SCI1BDH = 0;                   //1. Set Baud Rate to 9615 Hz

  SCI1BDL = 156;

 

  SCI1CR1 = 0b00100100;          // 7 - One pin operation

                                 // 6 - SCI Enabled in Wait Mode

                                 // 5 - RXD and TXD NOT connected internally

                                 // 4 - Eight Bit Data

                                 // 3 - Address mark wakeup

                                 // 2 - Idle character bit count beings after stop bit

                                 // 1 - No Parity

                                 // 0 - Even Parity (but it is disable)

 

  SCI1CR2 = 0b00101100;          // 7 - Transmit Interrupt Enabled

                                 // 6 - NO Transmit Complete Interrupt

                                 // 5 - Receive Interrupt Enabled

                                 // 4 - NO Idle Line Interrupt

                                 // 3 - TXD Enabled

                                 // 2 - RXD Enabled

                                 // 1 - NO Receiver Wake Up

                                 // 0 - Currently Don't Send Break

 

}

 

 

void interrupt _Vec_sci1 XBee_Interrupt(void){

  char i;

  unsigned char status,data;

  static unsigned char framelength;

  status = SCI1SR1;     //read in the status reg

  data = SCI1DRL;       //get the data will clear flags

 

  //RECEIVE interrupt

  if((status & _S12_RDRF) == _S12_RDRF) {

    receive_packet[receive_index] = data;

    if(receive_index == 0) {

      //just a temp thing until the actual framelength comes through

      framelength = 20;

    }

    if(receive_index == 2) {

      framelength = data;

    }

    receive_index++;

    if(receive_index == framelength+4) {

      receive_complete = 1;

      receive_index = 0;

      if(( ((receive_packet[7] & BIT1HI) == 0) && (receive_packet[3] == 0x81)) || (receive_packet[3] == 0x89)) {

        for(i=0;i<PACKET_LENGTH;i++) {

          receive_buffer[i] = receive_packet[i];

        }

      }

    }

  } else {

 

  //TRANFER interrrupt

    if(((status & _S12_TDRE) == _S12_TDRE)){

      SCI1DRL = transmit_packet[transmit_index];

      transmit_index++;

      if(transmit_index == PACKET_LENGTH) {

        //the transfer is done

        transmit_index = 0;

        //turn off tx clear interrupt

        SCI1CR2 &= ~_S12_TIE;

      }

    }

  }

}

 

 

 

void XBee_Transmit(unsigned char byte1, unsigned char byte2,unsigned char byte3) {

  unsigned char i;

  char junk;

  static char packetID = 1;

  transmit_packet[0] = XBEE_START_BYTE;      //start deliminator

  transmit_packet[1] = 0;                    //data frame length MSB

  transmit_packet[2] = DATA_BYTES + 5;      //data frame length LSB

  transmit_packet[3] = SEND_MESSAGE;        //send message API Identifier

  transmit_packet[4] = packetID;             //unique packet identifier, can be anything, but

  packetID++;                                //should be different for tx to tx

  transmit_packet[5] = BOAT_ADDRESS_MSB;

  transmit_packet[6] = BOAT_ADDRESS_LSB;

  transmit_packet[7] = 0x00;                  //options, should set to zero

  transmit_packet[8] = byte1;

  transmit_packet[9] = byte2;

  transmit_packet[10] = byte3;

 

  //now compute checksum of data frame (index 3 to end of data frame)

  transmit_packet[DATA_BYTES+8] = 0xff;

  for(i=3;i<DATA_BYTES+8;i++) {

    transmit_packet[DATA_BYTES+8] -= transmit_packet[i];

  }

  (void)printf("     sent packet: %x %x %x %x %x %x %x %x (%x %x %x) %x\r\n",

                transmit_packet[0],transmit_packet[1],transmit_packet[2],

                transmit_packet[3],transmit_packet[4],transmit_packet[5],

                transmit_packet[6],transmit_packet[7],transmit_packet[8],

                transmit_packet[9],transmit_packet[10],transmit_packet[11]);

  //initiate the transfer by sending first byte

  junk = SCI1SR1; // must read status register before any transmit

  junk = SCI1DRL; // must read data register before any transmit

  //turn on transfer reg clear interrupt

  SCI1CR2 |= _S12_TIE;

  SCI1DRL = transmit_packet[0];    //send first byte

  transmit_index = 1;

}

 

 

 

void XBee_Receive(void) {

  static unsigned int potID, crabsonboat, crabsonshore = 0;

  unsigned char framelength,checksum;

  unsigned char i;

  if(receive_complete == 0) {

    return;

  }

  /*(void)printf("r     received packet: %x %x %x %x %x %x %x %x %x %x %x %x\r\n",

              receive_buffer[0],receive_buffer[1],receive_buffer[2],

              receive_buffer[3],receive_buffer[4],receive_buffer[5],

              receive_buffer[6],receive_buffer[7],receive_buffer[8],

              receive_buffer[9],receive_buffer[10],receive_buffer[11]);

  (void)printf("                       %x %x %x %x %x %x %x %x %x %x %x %x\r\n",

              receive_buffer[12],receive_buffer[13],receive_buffer[14],

              receive_buffer[15],receive_buffer[16],receive_buffer[17],

              receive_buffer[18],receive_buffer[19],receive_buffer[20],

              receive_buffer[21],receive_buffer[22],receive_buffer[23]);

              */

  //determine data frame length

  framelength = receive_buffer[2];

  //check checksum

  if(receive_buffer[3] == 0x89) {

    switch(receive_buffer[5]){

      case 0x00:

        //(void)printf("Successfully sent packet Id: ");

        break;

      case 0x01:

        (void)printf("No ACK failure on sent packet Id: ");

        break;

      case 0x02:

        (void)printf("No airspace failure on sent packet Id: ");

        break;

      case 0x03:

        (void)printf("Purge failure on sent packet Id: ");

        break;

    }

    //print packet ID

    (void)printf("%x\r\n",receive_buffer[4]);

  }

    checksum = 0xff;

    for(i=3;i<3+framelength;i++) {

    checksum -= receive_buffer[i];

  }

  if(receive_buffer[3] == 0x81) {

    (void)printf("r     received packet: %x %x %x %x %x %x %x %x %x %x %x %x\r\n",

              receive_buffer[0],receive_buffer[1],receive_buffer[2],

              receive_buffer[3],receive_buffer[4],receive_buffer[5],

              receive_buffer[6],receive_buffer[7],receive_buffer[8],

              receive_buffer[9],receive_buffer[10],receive_buffer[11]);

    (void)printf("                       %x %x %x %x %x %x %x %x %x %x %x %x\r\n",

              receive_buffer[12],receive_buffer[13],receive_buffer[14],

             receive_buffer[15],receive_buffer[16],receive_buffer[17],

              receive_buffer[18],receive_buffer[19],receive_buffer[20],

              receive_buffer[21],receive_buffer[22],receive_buffer[23]);

    if((receive_buffer[7] & BIT1HI) == BIT1HI) {

      (void)printf("BBBBB\r\n",

      receive_buffer[4],receive_buffer[5]);

    } else {

      //(void)printf("T\r\n");

      (void)printf("TTTTT___Received targeted message from %x%x: %x %x %x\r\n",

          receive_buffer[4],receive_buffer[5],receive_buffer[8],receive_buffer[9],receive_buffer[10]);

      if(checksum != receive_buffer[3+framelength]) {

         (void)printf("XXXXXXX Checksum mismatch %x %x\r\n",checksum,receive_buffer[3+framelength]);

      }

      potID = (unsigned int)receive_buffer[8];

      //check team assignment

      switch(potID) {

        case 0x11:

          LCD_Team(1);

          crabsonshore = 0;

          crabsonboat = 0;

          waterlogged = 0;

          break;

        case 0x22:

          crabsonshore = 0;

          crabsonboat = 0;

          waterlogged = 0;

          LCD_Team(2);

          break;

        case 0x44:

          crabsonshore = 0;

          crabsonboat = 0;

          waterlogged = 0;

          LCD_Wait_for_Team();

          break;

      }

      //check for start season

      if(potID == 0xcc) {

        waterlogged = 0;

        LCD_Start();

        crabsonboat = 0;

        crabsonshore = 0;

      }

      //check to see if we are waterlogged

      if(potID == 0xdd) {

        waterlogged = 1;

        LCD_Waterlogged();

        //(void)printf("waterlogged\r\n");

        crabsonboat = 0;

      }

      //check to see if we are at drydock

      if((potID == 0xff) && (waterlogged == 1)) {

        waterlogged = 0;

        LCD_Fixed();

        //(void)printf("drydock\r\n");

      }

      //check to see if it is unloading dock

      if((potID == 0xee) && (waterlogged == 0)) {

        crabsonshore += crabsonboat;

        crabsonboat = 0;

        LCD_Unload(crabsonshore);

        //(void)printf("Crab Unloaded!\r\n");

      }

      //put the new info on the LCD screen

      if((potID >= 0xa1) && (potID <= 0xac)) {

        if(waterlogged == 0) {

          crabsonboat = (unsigned int)receive_buffer[10] * 256 + (unsigned int)receive_buffer[11];

          (void)printf("on boat %i\r\n",crabsonboat);

          if(crabsonboat > 299) {

            LCD_Full();

          }

          else {

            LCD_Update((potID & 0x000f),(unsigned int)receive_buffer[9],

                crabsonboat,crabsonshore);

          }

        }

      }

      if(potID == 0xa0) {

        waterlogged = 0;

        LCD_Verify((potID & 0x000f),(unsigned int)receive_buffer[9]);

      }

    }

  }

  for(i=0;i<PACKET_LENGTH;i++){

    receive_buffer[i] = 0;

  }

  receive_complete = 0;

}
