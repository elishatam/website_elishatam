#include "BITDEFS.H"

#include <htc.h>

__CONFIG(HS & UNPROTECT & WDTDIS & PWRTDIS & MCLRDIS & BORDIS & IESODIS & FCMDIS);

/*  HS = high speed resonator

    UNPROTECT = code is not protected

    WDTDIS =  Watchdog timer off

    PWRTDIS = Power up timer off

    MCLRDIS = Master Clear disabled

    BORDIS = Brownout disabled

    IESODIS = Internal external switch over mode disabled

    FCMDIS = Monitor Clock fail-safe disabled

*/

 

#define OFF 0

#define ON 1

 

/*----------- Module Prototypes ----------*/

void driveMO (char);

void driveM1 (char);

void updateMotors (void);

void init_Motors (void);

void init_SPI_SlaveMode (void);

void checkSPI (void);

void checkStepMotor (void);

 

/*-------------------- Global Variables ----------------------*/

//global variables for SPI

unsigned char SPI_timeout_cnt = 0x00;

unsigned char SPI_Cmd = 0x00;

 

//global variables for propel motor control

unsigned char M0_PWMStatus;

unsigned char M0_PWMtime;

unsigned char M0_dir;

 

unsigned char M1_PWMStatus;

unsigned char M1_PWMtime;

unsigned char M1_dir;

 

//global variables for stepper motor (crab pot indicator) control

unsigned char Motor_Steps[4] = {0b1010, 0b1001, 0b0101, 0b0110};

unsigned char stepIndex = 0;

unsigned char desStepperPos = 0;    //where do we want the stepper motor to be at

unsigned char curStepperPos = 0;    //where is the stepper motor currently at

const unsigned char cmdPerStep = 2; //#of cmd units per step pos (cmd = 100 => pos=100/cmdPerStep)

const unsigned char stepDelay = 3; // slows down steps by 5 ms * delay

 

unsigned char cmd;

unsigned char cmd_period;

unsigned char address;

unsigned char dir;

 

 

/*--------------------- Code ----------------------*/

void main (void){

   init_SPI_SlaveMode();

   init_Motors();

 

    while(1){

      updateMotors();

      checkSPI();

    }

}

 

void checkStepMotor (void) {

   static char curDelayStep = 0;

   if (curDelayStep >= stepDelay) {

      curDelayStep = 0;

      if (desStepperPos == curStepperPos) return;

 

      // need to step motor

      if (desStepperPos > curStepperPos) {

         stepIndex--;

         curStepperPos++;

      }

      else {

         stepIndex++;

         curStepperPos--;

      }

 

      // update outputs to new step

      stepIndex &= 0x03; // mod 4

      PORTC = (PORTC & 0xF0) | (Motor_Steps[stepIndex]);

   }

   else {

      curDelayStep++;

   }

}

 

void checkSPI (void) {

   if (BF != 0x01) { //check sync comm for byte

      if (SPI_timeout_cnt > 10) {

         SPI_timeout_cnt = 11;

         //turn off motors

         M0_PWMtime = 0;

         M1_PWMtime = 0;

      }

      return;

   }

   // reset timeout count

   SPI_timeout_cnt = 0;

 

   SPI_Cmd = SSPBUF;

   SSPBUF = 0x00;

 

   //process SPI_Cmd

   address = SPI_Cmd/64;

   if (address == 0b10) {

      // byte commands crab pot indicator

      cmd = (SPI_Cmd & BIT7LO & BIT6LO);

      if (cmd > 63) cmd = 63; // make sure to saturate at 100

      desStepperPos = (cmd*4)/5;

   }

   else {

      // byte commands one of the two propeller motors

      dir = (SPI_Cmd & BIT7LO & BIT6LO)/32;

      cmd_period = ((SPI_Cmd & BIT7LO & BIT6LO & BIT5LO)*65)/10;

 

      switch (address) {

         case 0x00:

            M0_PWMtime = cmd_period;

            M0_dir = dir;

            break;

         case 0x01:

            M1_PWMtime = cmd_period;

            M1_dir = dir;

            break;

      }

   }

}

 

void init_SPI_SlaveMode (void){

   char temp;

    //Set inputs/outputs

    ANSEL = 0;          //make all bits digital

    ANSELH = 0;

 

    TRISC &= BIT7LO;    //Set SDO as output

    TRISB |= BIT6HI;    //Set SCK as input

    TRISB |= BIT4HI;    //Set SDI as input

    TRISC |= BIT6HI;    //Set SS  as input

 

    //Set SPI config

    SSPSTAT = 0x00;     //see page 15-3 of PIC mannual

    //SSPCON  = 0x35;   // set up SPI port, Slave mode, No SS

    SSPCON = 0x34;      // set up SPI port, Slave mode, SS enabled

    temp = SSPBUF;

 

    //Set Timer0

    OPTION = 0b00000110; //pre-scale to 128; w/extrn osc @ 20M, 1 tick ~ 0.0256 [ms]

}

 

void init_Motors (void) {

   TRISC &= (BIT3LO & BIT2LO & BIT1LO & BIT0LO); //Set C3:0 as outputs for stepper motor cmds

   TRISC &= (BIT5LO & BIT4LO); // set C5:4 as outputs for Motor1

   TRISA &= (BIT1LO & BIT0LO); // set A1:0 as outputs for Motor0

 

   //turn off propeller motors

   M0_PWMStatus = OFF;

   M0_PWMtime = 0;

   M0_dir = 0;

 

   M1_PWMStatus = OFF;

   M1_PWMtime = 0;

   M1_dir = 0;

 

   //set ouputs on stepper motor

   PORTC = (PORTC & 0xF0) | (Motor_Steps[stepIndex]);

 

   //reset TMR0

   TMR0 = 0;

}

 

void updateMotors (void) {

   if (TMR0 > 200) {

      //increment timeout count

      SPI_timeout_cnt++;

      //reset timer 0

      TMR0 = 0;

      //make both motors powered

      driveMO(0x01);

      driveM1(0x01);

      M0_PWMStatus = ON;

      M1_PWMStatus = ON;

      //check if step motor needs to be stepped

      checkStepMotor();

   }

 

   if (M0_PWMStatus == ON) {

      if (TMR0 > M0_PWMtime) {

         //turn off motor 0

         driveMO(0x00);

         M0_PWMStatus = OFF;

      }

   }

 

   if (M1_PWMStatus == ON) {

      if (TMR0 > M1_PWMtime) {

         //turn off motor 1

         driveM1(0x00);

         M1_PWMStatus = OFF;

      }

   }

}

 

void driveMO (char on) {

   // A1, A0

   if ((on == 0x00) || (M0_PWMtime == 0)) {

      //turn off motor

      if (M0_dir == 0x00) {

         PORTA &= BIT1LO;

         PORTA &= BIT0LO;

      }

      else {

         PORTA |= BIT1HI;

         PORTA |= BIT0HI;

      }

   }

   else {

      //drive motor

      if (M0_dir == 0x00) {

         PORTA |= BIT1HI;

         PORTA &= BIT0LO;

      }

      else {

         PORTA &= BIT1LO;

         PORTA |= BIT0HI;

      }

   }

}

 

void driveM1 (char on) {

   // C5, C4

   if ((on == 0x00) || (M1_PWMtime == 0)) {

      //turn off motor

      if (M1_dir == 0x00) {

         PORTC &= BIT5LO;

         PORTC &= BIT4LO;

      }

      else {

         PORTC |= BIT5HI;

         PORTC |= BIT4HI;

      }

   }

   else {

      //drive motor

      if (M1_dir == 0x00) {

         PORTC |= BIT5HI;

         PORTC &= BIT4LO;

      }

      else {

         PORTC &= BIT5LO;

         PORTC |= BIT4HI;

      }

   }

}
