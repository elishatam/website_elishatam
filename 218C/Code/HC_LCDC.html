LCD Display Module

 

Creates two interface functions; one that will initialize the LCD screen

and another to write a character to the LCD screen.

 

It uses the TMRS12_ functions provided in timers12.h

and the BITxxx variables provided in ME218_C32.h

 

***********

Wiring assignments:

***********

LCD pin#

1 (GRD)    -> wired to ground

2 (VCC)    -> wired to +5V voltage supply

3 (VC)     -> wired to ~10k potentiometer attached to 5V voltage supply

4 (RegSel) -> wired to U0

5 (R/W)    -> wired to ground (write only)

6 (ENABLE) -> wired to +5V

7 (DB0)    -> wired to (T0)

8 (DB1)    -> wired to (T1)

9 (DB2)    -> wired to (T2)

10 (DB3)   -> wired to (T3)

11 (DB4)   -> wired to (T4)

12 (DB5)   -> wired to (T5)

13 (DB6)   -> wired to (T6)

14 (DB7)   -> wired to (T7)

***********/

static void Select_Data_Register(void){

  PTU |= BIT0HI;

}

 

static void Select_Command_Register(void){

  PTU &= BIT0LO;

  }

 

static void Pulse_Enable(void) {

  PTU |= BIT1HI;

  wait(1);

  PTU &= BIT1LO;

}

 

void LCD_Init(void){

  (void)printf("LCD init\r\n");

 /*

 prepares LCD for display, clears screen, sets cursor position and type

 sets scoll mode and cursor shift mode

 */

  wait(40);           //wait for LCD screen to power up

  Select_Command_Register();

  //function set, 2 line mode, 5x10 characters

  PTT = 0b00111100;

  Pulse_Enable();

  wait(2);

  PTT = 0b00111100;

  Pulse_Enable();

  wait(1);

  PTT = 0b00111100;

  Pulse_Enable();

  wait(1);

  //cursor off, display on

  PTT = 0b00001100;

  Pulse_Enable();

  wait(1);

  //display clear

  PTT = 0b00000001;

  Pulse_Enable();

  wait(2);

  //entry mode set

  PTT = 0b00000110;

  Pulse_Enable();

  wait(1);

  //shift

  PTT = 0b00010000;

  Pulse_Enable();

  wait(1);

  //just to get it set up in the right place

  Put_Char(' ');

}

 

 

void Put_Char(unsigned char letter) {

  Select_Data_Register();

  //letter is expressed in hex

  PTT = letter;

  Pulse_Enable();

}

void LCD_Fixed(void) {

  unsigned char i;

  unsigned char fixed[]    = "     Boat fixed!    ";

  unsigned char crabon[]   = "     CRAB ON!!!!    ";

  TIM0_TIE  &= ~_S12_C5I;                           //disable blink interrupt

  Skip_Line();

  for(i=0;i<20;i++){

    Put_Char(crabon[i]);

  }

  for(i=0;i<20;i++){

    Put_Char(fixed[i]);

  }

  Skip_Line();

}
